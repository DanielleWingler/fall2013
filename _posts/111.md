from turtle import *
import time
 
color("green")

up()

goto(0,-50)

down()

circle(50)
up()

goto(0,0)
down()

for deg in range(0, 61, 6):

    forward(100)
    right(90)
    forward(100)
    right(90)
    forward(100)
    right(90)
    forward(100)
     
up()
goto(-150,-120)
color("red")
write("Done!")
